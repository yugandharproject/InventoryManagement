CREATE TABLE `inventory_product` (
	`ID_PK` VARCHAR(50) NOT NULL COLLATE 'utf8_bin',
	`VERSION` DECIMAL(22,0) NOT NULL,
	`CREATED_TS` DATETIME(6) NOT NULL,
	`DELETED_TS` DATETIME(6) NULL DEFAULT NULL,
	`UPDATED_TS` DATETIME(6) NOT NULL,
	`UPDATED_BY_USER` VARCHAR(50) NOT NULL COLLATE 'utf8_bin',
	`UPDATED_BY_TXN_ID` VARCHAR(100) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`PRODUCT_ID` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`Supplier_ID` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`Product_Name` VARCHAR(30) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`Product_Type` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`Product_Category` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`HSN-SAC Code` VARCHAR(10) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`Product_Status` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`Product_Expiry_Date` DATETIME(6) NULL DEFAULT NULL,
	`Product_Arrival_Date` DATETIME(6) NULL DEFAULT NULL,
	`Quantity_Recevied` DECIMAL(22,0) NULL DEFAULT NULL,
	`Quantity_Inhand` DECIMAL(22,0) NULL DEFAULT NULL,
	`Quantity_UnitOfMeasure` VARCHAR(50) NULL DEFAULT NULL,
	`Bill_number` VARCHAR(100) NULL DEFAULT NULL COLLATE 'utf8_bin',
	PRIMARY KEY (`ID_PK`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;



CREATE TABLE `ref_UnitOfMeasure` (
	`ID_PK` VARCHAR(50) NOT NULL COLLATE 'utf8_bin',
	`VERSION` DECIMAL(22,0) NOT NULL,
	`CREATED_TS` DATETIME(6) NOT NULL,
	`DELETED_TS` DATETIME(6) NULL DEFAULT NULL,
	`UPDATED_TS` DATETIME(6) NOT NULL,
	`UPDATED_BY_USER` VARCHAR(50) NOT NULL COLLATE 'utf8_bin',
	`UPDATED_BY_TXN_ID` VARCHAR(100) NULL DEFAULT NULL COLLATE 'utf8_bin',
	`CONFIG_LANGUAGE_CODE_KEY` VARCHAR(50) NOT NULL COLLATE 'utf8_bin',
	`KEY` VARCHAR(50) NOT NULL COLLATE 'utf8_bin',
	`VALUE` VARCHAR(50) NOT NULL COLLATE 'utf8_bin',
	`DESCRIPTION` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8_bin',
	PRIMARY KEY (`ID_PK`),
	UNIQUE INDEX `REF_UOM_UNIQUEKEY` (`KEY`, `CONFIG_LANGUAGE_CODE_KEY`),
	INDEX `INDX_REF_UOM1` (`KEY`),
	INDEX `INDX_REF_UOM2` (`VALUE`)
)
COLLATE='utf8_bin'
ENGINE=InnoDB
;


CREATE TABLE IF NOT EXISTS `INVENTORY_ORDER` (
  `ID_PK` varchar(50) COLLATE utf8_bin NOT NULL,
  `VERSION` decimal(22,0) NOT NULL,
  `CREATED_TS` datetime(6) NOT NULL,
  `DELETED_TS` datetime(6) DEFAULT NULL,
  `UPDATED_TS` datetime(6) NOT NULL,
  `UPDATED_BY_USER` varchar(50) COLLATE utf8_bin NOT NULL,
  `UPDATED_BY_TXN_ID` varchar(100) COLLATE utf8_bin DEFAULT NULL,
  `ORDER_ID` varchar(50) COLLATE utf8_bin,
  `PRODUCT_ID` varchar(50) COLLATE utf8_bin,
  `UNIT_PRICE`  decimal(22,0) NOT NULL,
  `UNIT_QUANTITY` decimal(22,0) NOT NULL
  );
